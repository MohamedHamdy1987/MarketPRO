/* ============================================================ Market Pro – invoices.js (FINAL – Consignment Logic) ============================================================ */ import { supabase, ensureUser, dbInsert, dbUpdate, dbDelete } from "../data.js"; import { toast, inputModal, formatCurrency, formatDate, emptyState } from "../ui.js"; /* ───────────────────────────────────────── عرض الفواتير ───────────────────────────────────────── */ export async function renderInvoicesPage(app){ const user = await ensureUser(); const { data: invoices } = await supabase .from("invoices") .select("*") .eq("user_id", user.id) .order("date",{ ascending:false }); app.innerHTML = `
📄 الفواتير
${ !(invoices||[]).length ? emptyState('📄','لا توجد فواتير','') : invoices.map(inv=>`
${inv.supplier_name}
${formatDate(inv.date)}
${inv.status}
`).join('') } `; } /* ───────────────────────────────────────── إضافة فاتورة ───────────────────────────────────────── */ window.openAddInvoice = function(){ inputModal({ title:'فاتورة جديدة', fields:[ {id:'supplier_name', label:'اسم المورد', required:true} ], onSubmit: async(vals)=>{ await dbInsert("invoices",{ supplier_name: vals.supplier_name, status: "draft", date: new Date().toISOString() }); toast("تم إنشاء الفاتورة","success"); navigate('invoices'); } }); }; /* ───────────────────────────────────────── عرض الفاتورة ───────────────────────────────────────── */ window.openInvoice = async function(invoiceId){ const app = document.getElementById("app"); const { data: invoice } = await supabase .from("invoices") .select("*") .eq("id", invoiceId) .single(); const { data: products } = await supabase .from("invoice_products") .select("*") .eq("invoice_id", invoiceId); const total = (products||[]).reduce((s,p)=>s+Number(p.qty||0),0); const sold = (products||[]).reduce((s,p)=>s+Number(p.sold||0),0); app.innerHTML = `
🚚 ${invoice.supplier_name}
الكمية ${total} • مباع ${sold}
${ invoice.status === "draft" ? ` ` : `معتمدة` }
📦 الأصناف ${ invoice.status === "draft" ? `` : '' }
${ !(products||[]).length ? "لا توجد أصناف" : products.map(p=>`
${p.name} ${p.qty} / ${p.sold||0}
`).join('') }
`; }; /* ───────────────────────────────────────── إضافة صنف ───────────────────────────────────────── */ window.addProduct = function(invoiceId){ inputModal({ title:'إضافة صنف', fields:[ {id:'name', label:'اسم الصنف', required:true}, {id:'qty', label:'الكمية', type:'number', required:true} ], onSubmit: async(vals)=>{ await dbInsert("invoice_products",{ invoice_id: invoiceId, name: vals.name, qty: Number(vals.qty), sold: 0 }); toast("تم إضافة الصنف","success"); openInvoice(invoiceId); } }); }; /* ───────────────────────────────────────── اعتماد الفاتورة ───────────────────────────────────────── */ window.confirmInvoice = async function(invoiceId){ try{ const { data: products } = await supabase .from("invoice_products") .select("*") .eq("invoice_id", invoiceId); if(!products.length){ throw new Error("لا توجد أصناف"); } await dbUpdate("invoices", invoiceId,{ status: "confirmed" }); toast("تم اعتماد الفاتورة ✅","success"); openInvoice(invoiceId); }catch(err){ toast(err.message,"error"); } };
